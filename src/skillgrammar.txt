skillgraph
  = BOM? line*

BOM
  = '\uFEFF'

line
  = node 
  / comment

node
  = name '.' NL 
  / name ':' name number (',' name number)* '.' 

NL
  = '\r\n'

comment
  = '#' .*

name
  = [a-z]+

number
   = [0-9]+

_ "whitespace"
  = whitespace*

// Whitespace is undefined in the original JSON grammar, so I assume a simple
// conventional definition consistent with ECMA-262, 5th ed.
whitespace
  = [ \t\n\r]
