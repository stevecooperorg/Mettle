skillgraph
  = BOM? line*

BOM
  = '\uFEFF'

line
  = node 
  / comment

node
  = _ name _ '.' _ NL 
  / _ name _ ':' _ name _ number _ (',' _ name _ number)* _ '.' 

NL
  = '\r' '\n'

comment
  = '#' .* NL

name
  = [a-z]+

number
   = [0-9]+

_ "whitespace"
  = whitespace*

// Whitespace is undefined in the original JSON grammar, so I assume a simple
// conventional definition consistent with ECMA-262, 5th ed.
whitespace
  = [ \t]
